# Write a program to accept required details from the user to create a resume text file. Open the file in the editor for displaying / print

def inputinfo():
    name = input("Enter your name: ")
    age = input("Enter your age: ")
    address = input("Enter your address: ")
    phone = input("Enter your phone number: ")
    email = input("Enter your email: ")
    education = input("Enter your education: ")
    skills = input("Enter your skills: ")
    experience = input("Enter your experience: ")
    with open("resume.txt", "w") as f:
        f.write("==============================================================\n")
        f.write("\t\t\tResume Details\n")
        f.write("==============================================================\n")
        f.write("Name: " + name.capitalize() + "\n")
        f.write("Age: " + age + "\n")
        f.write("Address: " + address + "\n")
        f.write("Phone: +91 " + phone[0:5] +" "+ phone[5:11]+ "\n")
        f.write("Email: " + email + "\n")
        f.write("Education: " + education.upper() + "\n")
        f.write("Skills: " + skills + "\n")
        f.write("Experience: " + experience + "\n")
        f.write("==============================================================\n")
    f.close()
    
def display():
    with open("resume.txt", "r") as f:
        print(f.read())

def main():
    inputinfo()
    display()

main()

balance = 5000

def bank():

    account = input()

    pin = input()
    if account == "12345" and pin == "1234":
        print("You are now logged in")

            print("3. Deposit funds")
            print("4. Exit")
            option = input()
            if option == "1":
                check_balance()
            elif option == "2":
                withdraw_bal()
            elif option == "3":
                deposit_bal()

def check_balance():
    with open("bank.csv", "r") as f:
        for line in f:
            if "12345" in line:
                balance = line.split(",")[3]
                print("Your balance is: " + balance)
    f.close()

def withdraw_bal():
    print("Enter the account number: ")
    account = input()
    print("Enter the amount to withdraw: ")
    withdraw = input()
    with open("bank.csv", "r") as f:
        for line in f:
            if account in line:
                balance = line.split(",")[3]
                if int(withdraw) <= int(balance):
                    with open("bank.csv", "r") as f:
                        for line in f:
                            if account in line:
                                balance = line.split(",")[3]
                                balance = int(balance) - int(withdraw)
                                print("Your new balance is: " + str(balance))
                    f.close()
                    with open("bank.csv","w") as f:
                        f.write(line.replace(line.split(",")[3], str(balance)))
                    f.close()
                else:
                    print("You do not have enough funds")
                    
def deposit_bal():
    print("Enter the account number: ")
    account = input()
    print("Enter the amount to deposit: ")
    deposit = input()
    with open("bank.csv", "r") as f:
        for line in f:
            if account in line:
                balance = line.split(",")[3]
                with open("bank.csv", "r") as f:
                    for line in f:
                        if account in line:
                            balance = line.split(",")[3]
                            balance = int(balance) + int(deposit)
                            print("Your new balance is: " + str(balance))
                f.close()
                with open("bank.csv", "w") as f:
                    f.write(line.replace(line.split(",")[3], str(balance)))
                f.close()
