#mapping
emails=['pooja@', '@divya', 'karthik@gmail', 'sai@gmail.com', 'gautham@hotmail.com']
def valid_mail(mail):
    if(mail.find('@') > -1 and mail.endswith(".com") and mail[0].isalpha()):
        return True
    else:
        return False

op = (map(valid_mail, emails))
for x in op:
    print(x)

#decorators
def calculate(func):
    def inner(x,y):
        if(x == 0 or y == 0):
            print("Invalid input")
            return 0
        else:
            return func(x,y)
    return inner

@calculate
def add(x,y):
    return x + y
@calculate
def sub(x,y):
    return x - y
@calculate
def mul(x,y):
    return x * y
@calculate
def div(x,y):
    return x/y
print(div(5,6))

#NROM
from functools import reduce
import math
m = [[1,2,3],[4,5,6],[7,8,9]]
def elements():
    a = []
    for x in range(len(m)):
        for y in range(len(m[x])):
            a.append(m[x][y])
    return a
x = math.sqrt(reduce(lambda a,b : a + b,map(lambda x: x * x,elements())))
print("%0.2f"%x)
